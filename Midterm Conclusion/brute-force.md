### 3. Pseudocode
```
procedure bubbleSort(A)
    n тЖР length(A)
    repeat
        swapped тЖР false
        for i тЖР 0 to n-2 do
            if A[i] > A[i+1] then
                swap A[i], A[i+1]
                swapped тЖР true
        n тЖР n - 1
    until swapped = false
end procedure
```

### 4. Python Example
```python
def bubble_sort(arr):
    n = len(arr)
    for i in range(n):
        swapped = False
        for j in range(0, n - i - 1):
            if arr[j] > arr[j + 1]:
                arr[j], arr[j + 1] = arr[j + 1], arr[j]
                swapped = True
        if not swapped:
            break
    return arr

# Example usage
data = [5, 3, 8, 4, 2]
sorted_data = bubble_sort(data)
print(sorted_data)  # Output: [2, 3, 4, 5, 8]
```

### 3. Pseudocode
```

    for i тЖР 0 to length(A)-2 do
        minIndex тЖР i
        for j тЖР i+1 to length(A)-1 do

end procedure
```

### 4. Python Example
```python
def selection_sort(arr):
    n = len(arr)
    for i in range(n):
        min_idx = i
        for j in range(i+1, n):
            if arr[j] < arr[min_idx]:
                min_idx = j
        arr[i], arr[min_idx] = arr[min_idx], arr[i]
    return arr

# Example usage
data = [5, 3, 8, 4, 2]
sorted_data = selection_sort(data)
print(sorted_data)  # Output: [2, 3, 4, 5, 8]
```
